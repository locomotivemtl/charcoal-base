<?php

namespace Charcoal\Object;

// Dependency from 'charcoal-config'
use \Charcoal\Config\AbstractConfig;

/**
 * Configuration set for the {@see \Charcoal\Object\RoutableTrait}.
 */
class RoutableObjectConfig extends AbstractConfig
{
    /**
     * Retrieve the default settings for configuring a routable object.
     *
     * @return array
     */
    public function defaults()
    {
        return [
            'pattern'     => null,
            'fields'      => [],
            'updatable'   => true,
            'unique'      => true,
            'unique_base' => null,
            'separator'   => '-',
            'prefix'      => '',
            'suffix'      => '',
            'style'       => 'default',
            'date_format' => 'Y-m-d-H:i',
            'handlers'    => []
        ];
    }
}
